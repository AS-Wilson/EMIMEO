		\pattern[pattern=north east lines,pattern color=red!45] (c3) -- (c1) -- (c2) -- (c4) -- cycle;

		\draw[{Circle[length=1mm]}->,shorten <=-0.5mm,color=SciencePurple] (\xmin+0.25,-3.4) to [bend left = 7.5] (\xmin+0.25,\ymin); % Transition TWO to ONE
